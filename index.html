Even Odd in an given array

const findEvenOdd = (N, Arr) => 
    {
        let B=[];
        var a = Arr.filter((x) => x%2 === 0);
        var b = a.reduce((acc,sum) => acc = acc+sum,0);
        B.push(b);
        
        var c = Arr.filter((x) => x%2 !== 0);
        var d = c.reduce((acc,sum) => acc = acc+sum,0);
        B.push(d);
        
        return B;
    };

Birthday Game

function Birthday_Game(arr,D ,M) {
    let counter = 0;
    for(let i = 0; i<= arr.length - 1; i++){
      let subArray = arr.slice(i, i+M);
      if(subArray.length === M){
        let sum = subArray.reduce((a,b)=> a+b,0);
        
        if(sum === D){
          counter += 1;
        }
      }
    }
    return counter;
   
   }
